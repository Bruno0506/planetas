from picamera2 import Picamera2
from PIL import Image
import numpy as np

# Inicializar la cámara
picam2 = Picamera2()
picam2.start()  # Iniciar la cámara

# Capturar una imagen
image = picam2.capture_array()

# Verifica si la imagen se capturó correctamente
if image is not None:
    # Convertir la imagen a un formato que pueda mostrar PIL
    image_pil = Image.fromarray(image)

    # Mostrar la imagen
    image_pil.show()  # Esto abrirá la imagen en el visor de imágenes predeterminado
else:
    print("Error: No se pudo capturar la imagen.")

picam2.close()  # Cerrar la cámara al finalizar
